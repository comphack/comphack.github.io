196 void LobbyServer::SendWorldList(const std::shared_ptr<
197  libcomp::TcpConnection>& connection) const
198 {
199  libcomp::Packet p;
200  p.WritePacketCode(LobbyToClientPacketCode_t::PACKET_WORLD_LIST);
201 
202  auto worlds = GetWorlds();
203  worlds.remove_if([](const std::shared_ptr<lobby::World>& world)
204  {
205  return world->GetRegisteredWorld()->GetStatus()
206  == objects::RegisteredWorld::Status_t::INACTIVE;
207  });
208 
209  // World count.
210  p.WriteU8((uint8_t)worlds.size());
211 
212  // Add each world to the list.
213  for(auto world : worlds)
214  {
215  auto worldServer = world->GetRegisteredWorld();
216 
217  // ID for this world.
218  p.WriteU8(worldServer->GetID());
219 
220  // Name of the world.
221  p.WriteString16Little(libcomp::Convert::ENCODING_UTF8,
222  worldServer->GetName(), true);
223 
224  auto channels = world->GetChannels();
225 
226  // Number of channels on this world.
227  p.WriteU8((uint8_t)channels.size());
228 
229  // Add each channel for this world.
230  for(auto channel : channels)
231  {
232  // Name of the channel. This used to be displayed in the channel
233  // list that was hidden from the user.
234  p.WriteString16Little(libcomp::Convert::ENCODING_UTF8,
235  channel->GetName(), true);
236 
237  // Ping time??? Again, something that used to be in the list.
238  p.WriteU16Little(1);
239 
240  // 0 - Visible | 2 - Hidden (or PvP)
241  // Pointless without the list.
242  p.WriteU8(0);
243  }
244  }
245 
246  if(nullptr == connection)
247  {
248  // Send to all client connections
249  auto connections = mManagerConnection->GetClientConnections();
250  libcomp::TcpConnection::BroadcastPacket(connections, p);
251  }
252  else
253  {
254  connection->SendPacket(p);
255  }
256 }
