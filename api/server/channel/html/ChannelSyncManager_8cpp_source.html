240  const std::shared_ptr<libcomp::Object>& obj, bool isRemove,
241  const libcomp::String& source)
242 {

244  (void)source;
245 
246  bool success = false;
247 
248  auto entry = std::dynamic_pointer_cast<objects::SearchEntry>(obj);
249 
250  auto& entryList = mSearchEntries[entry->GetType()];
251 
252  auto it = entryList.begin();

254  {
255  if((*it)->GetEntryID() == entry->GetEntryID())
256  {
257  it = entryList.erase(it);
258  if(!isRemove)
259  {
260  // Replace the existing element
261  entryList.insert(it, entry);
262  }
263 
264  success = true;
265  break;
266  }
267 
268  it++;
269  }
270 
271  if(!success)
272  {
273  if(isRemove)
274  {
275  LOG_WARNING(libcomp::String("No SearchEntry with ID '%1' found"
276  " for sync removal\n").Arg(entry->GetEntryID()));
277  }
278  else
279  {
280  entryList.push_front(entry);
281 
282  // Re-sort by entry ID, highest first
283  entryList.sort([](
284  const std::shared_ptr<objects::SearchEntry>& a,
285  const std::shared_ptr<objects::SearchEntry>& b)
286  {
287  return a->GetEntryID() > b->GetEntryID();
288  });
289 
290  success = true;
291  }
292  }
293 
294  if(success)
295  {
296  bool isApp = (int8_t)entry->GetType() % 2 == 1;
297 
298  std::shared_ptr<objects::SearchEntry> parent;
299  if(isApp)
300  {
301  auto parentType = (objects::SearchEntry::Type_t)(
302  (int8_t)entry->GetType() - 1);
303  for(auto entry2 : mSearchEntries[parentType])
304  {
305  if(entry2->GetEntryID() == entry->GetParentEntryID())
306  {
307  parent = entry2;
308  break;
309  }
310  }
311  }
312 
313  // If an app is being removed, inform both characters involved, otherwise
314  // just inform the source character
315  std::set<int32_t> cids = { entry->GetSourceCID() };
316  if(isRemove && parent)
317  {
318  cids.insert(parent->GetSourceCID());
319  }
320 
321  auto server = mServer.lock();
322  for(int32_t cid : cids)
323  {
324  auto client = server->GetManagerConnection()->GetEntityClient(
325  cid, true);
326  if(client)
327  {
328  int32_t removeReason = 0;
329  auto packetCode = ChannelToClientPacketCode_t::PACKET_SEARCH_ENTRY_REGISTER;
330  switch(entry->GetLastAction())
331  {
332  case objects::SearchEntry::LastAction_t::UPDATE:
333  // Only clan search entries are ever actually updated, all others are re-registered
334  if(entry->GetType() != objects::SearchEntry::Type_t::CLAN_JOIN &&
335  entry->GetType() != objects::SearchEntry::Type_t::CLAN_RECRUIT)
336  {
337  packetCode = ChannelToClientPacketCode_t::PACKET_SEARCH_ENTRY_UPDATE;
338  }
339  break;
340  case objects::SearchEntry::LastAction_t::REMOVE_MANUAL:
341  packetCode = ChannelToClientPacketCode_t::PACKET_SEARCH_ENTRY_REMOVE;
342  break;
343  case objects::SearchEntry::LastAction_t::REMOVE_LOGOFF:
344  packetCode = ChannelToClientPacketCode_t::PACKET_SEARCH_ENTRY_REMOVE;
345  removeReason = 6;
346  break;
347  case objects::SearchEntry::LastAction_t::REMOVE_EXPIRE:
348  packetCode = ChannelToClientPacketCode_t::PACKET_SEARCH_ENTRY_REMOVE;
349  removeReason = 1;
350  break;
351  case objects::SearchEntry::LastAction_t::REMOVE_SPECIAL:
352  packetCode = ChannelToClientPacketCode_t::PACKET_SEARCH_ENTRY_REMOVE;
353  removeReason = 2;
354  break;
355  case objects::SearchEntry::LastAction_t::ADD:
356  default:
357  break;
358  }
359 
360  libcomp::Packet reply;
361  reply.WritePacketCode(packetCode);
362  reply.WriteS32Little((int32_t)entry->GetType());
363 
364  if(packetCode == ChannelToClientPacketCode_t::PACKET_SEARCH_ENTRY_REGISTER)
365  {
366  reply.WriteS32Little(0); // Success
367  reply.WriteS32Little(entry->GetEntryID());
368  }
369  else
370  {
371  reply.WriteS32Little(entry->GetEntryID());
372  reply.WriteS32Little(0); // Success
373 
374  if(packetCode == ChannelToClientPacketCode_t::PACKET_SEARCH_ENTRY_REMOVE)
375  {
376  reply.WriteS32Little(removeReason);
377  }
378  }
379 
380  client->SendPacket(reply);
381  }
382  }
383 
384  // If the type is odd, it is an application so we need to notify
385  // the person being replied to if they are on this channel
386  if(isApp && entry->GetLastAction() == objects::SearchEntry::LastAction_t::ADD &&
387  parent)
388  {
389  auto client = parent ? server->GetManagerConnection()->GetEntityClient(
390  parent->GetSourceCID(), true) : nullptr;
391  if(client)
392  {
393  libcomp::Packet notify;
394  notify.WritePacketCode(ChannelToClientPacketCode_t::PACKET_SEARCH_APPLICATION);
395  notify.WriteS32Little((int32_t)parent->GetType());
396  notify.WriteS32Little(parent->GetEntryID());
397  notify.WriteS32Little(entry->GetEntryID());
398 
399  client->SendPacket(notify);
400  }
401  }
402  }
403 
404  return success ? SYNC_UPDATED : SYNC_FAILED;
405 }
