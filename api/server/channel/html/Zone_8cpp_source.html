998  // the most significant restrictions first

1009  dateActive = pair.first <= dateSum &&
1010  dateSum <= pair.second;
1011  }
1012  else

1014  // Rollover compare
1015  dateActive = pair.first <= dateSum ||
1016  dateSum <= pair.second;
1017  }

1022  }

1028  }
1029  }

1032  {
1033  return false;

1037  {
1038  bool timeActive = false;

1055
