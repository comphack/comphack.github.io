361  }
362  }
363  }

393  return false;
394  }
395  }
396

398  {
399  case AICommandType_t::MOVE:

401  {
402  if(eState->IsMoving())
403  {
404  return false;

406  else

424  }
425  }

434  }
435  break;

437  {
438  // Do nothing if hit stunned or still charging
439  if(eState->GetStatusTimes(STATUS_HIT_STUN) ||
440  eState->GetStatusTimes(STATUS_KNOCKBACK) ||
441  eState->GetStatusTimes(STATUS_CHARGING))
442  {
443  return false;
444  }

447

449  auto skillManager = server->GetSkillManager();
450 
451  bool valid = true;
452

460  // Target invalid or dead, cancel the skill and move on

462  {
463  skillManager->CancelSkill(eState, activated->GetActivationID());
464  }
465  valid = false;
466  }
467  }
468

472  ctx->FreeCast = true;
473 
474  if(activated)
475  {
476  // Execute the skill
477  if(skillManager->ExecuteSkill(eState, activated->GetActivationID(),
478  activated->GetTargetObjectID(), ctx))
479  {
480  // Queue a wait command
481  QueueWaitCommand(aiState, (uint32_t)(aiState->GetThinkSpeed()
482  * RNG(int32_t, 1, 3)));
483  }
484  else
485  {
487  skillManager->CancelSkill(eState, activated->GetActivationID());
488  }
489  }
490  else
491  {
492  // Activate the skill
493  skillManager->ActivateSkill(eState, cmdSkill->GetSkillID(),
494  cmdSkill->GetTargetObjectID(), cmdSkill->GetTargetObjectID(), ctx);
495  }
496  }
497 
498  aiState->PopCommand();
499  }
500  break;
501  case AICommandType_t::SCRIPTED:
502  {
503  // Execute a custom scripted command
504  auto cmdScript = std::dynamic_pointer_cast<AIScriptedCommand>(cmd);
505 
506  int32_t result = 0;
507  if(!ExecuteScriptFunction(eState, cmdScript->GetFunctionName(), now,
508  result))
509  {
510  // Pop the command and move on
511  aiState->PopCommand();
512  }
513  else if(result == 0)
514  {
515  return false;
516  }
517  else
518  {
519  aiState->PopCommand();
520  if(result == 1)
521  {
522  return true;
523  }
524  }
525  }
526  break;
527  case AICommandType_t::NONE:
528  default:
529  aiState->PopCommand();
530  break;
531  }
532  }

813  std::pair<float, float> aggroCast(aiState->GetAggroValue(2, false, 2000.f),
814  aiState->GetAggroValue(2, true, 1.395f));
815

817  // leaving in doubles for higher chances when closer)
818  bool castingOnly = true;
819

821  for(auto aggro : { aggroCast, aggroNormal })
822  {

824  sourceY, (double)aggro.first);
825

827  filtered.remove_if([eState, castingOnly](
828  const std::shared_ptr<ActiveEntityState>& entity)
829  {
830  return entity->GetFaction() == eState->GetFaction() ||
831  (castingOnly && !entity->GetStatusTimes(STATUS_CHARGING)) ||
832  !entity->Ready();
833  });
834

836  // filter them out now
837  if(aggroLevelLimit < 99)
838  {
839  filtered.remove_if([aggroLevelLimit](
840  const std::shared_ptr<ActiveEntityState>& entity)
841  {
842  auto cs = entity->GetCoreStats();
843  return !cs || (int32_t)cs->GetLevel() > aggroLevelLimit;
844  });

852  entity->RefreshCurrentPosition(now);
853  }

856  for(auto fovEntity : ZoneManager::GetEntitiesInFoV(filtered, sourceX,
857  sourceY, eState->GetCurrentRotation(), aggro.second))
858  {
859  inFoV.push_back(fovEntity);
860  }
861  }
862 
863  castingOnly = false;
864  }
865 
866  if(inFoV.size() > 0)
867  {

870  {
871  // Possible target found, check line of sight
872  bool add = true;
873  if(geometry)
874  {
875  Line path(Point(sourceX, sourceY),
876  Point(entity->GetCurrentX(), entity->GetCurrentY()));
877 
878  Point collidePoint;
879  add = !geometry->Collides(path, collidePoint);
880  }
881 
882  if(add)
883  {
884  possibleTargets.push_back(entity);
885 
886  if(aiState->GetStatus() == AIStatus_t::WANDERING)
887  {
888  aiState->SetStatus(AIStatus_t::AGGRO);
889  }
890  }
891  }
892  }
893  }
894 
895  if(possibleTargets.size() > 0)
896  {
898  auto targetIter = possibleTargets.begin();
899  size_t randomIdx = (size_t)RNG(int32_t, 0,
900  (int32_t)(possibleTargets.size() - 1));
901  std::advance(targetIter, randomIdx);
902 
903  target = *targetIter;
904 
905  aiState->SetTarget(target ? target->GetEntityID() : 0);
906  }
907 
908  return target;

911 void AIManager::RefreshSkillMap(const std::shared_ptr<ActiveEntityState>& eState,
912  const std::shared_ptr<AIState>& aiState)
913 {

915  {
916  auto isEnemy = eState->GetEntityType() ==
917  objects::EntityStateObject::EntityType_t::ENEMY;
918

920  auto definitionManager = mServer.lock()->GetDefinitionManager();
921 
922  auto currentSkills = eState->GetCurrentSkills();

924  {
925  auto skillData = definitionManager->GetSkillData(skillID);
926 
927  // Active skills only
928  auto category = skillData->GetCommon()->GetCategory();
929  if(category->GetMainCategory() != 1) continue;
930 
931  auto range = skillData->GetRange();
932 
933  int16_t targetType = -1;
934  switch(range->GetValidType())
935  {
936  case objects::MiEffectiveRangeData::ValidType_t::ALLY:
937  targetType = (int16_t)AI_SKILL_TYPES_ALLY;
938  break;
939  case objects::MiEffectiveRangeData::ValidType_t::SOURCE:
940  targetType = (int16_t)AI_SKILL_TYPE_DEF;
941  break;
942  case objects::MiEffectiveRangeData::ValidType_t::ENEMY:
943  targetType = (int16_t)AI_SKILL_TYPES_ENEMY;
944  break;
945  case objects::MiEffectiveRangeData::ValidType_t::PARTY:
946  case objects::MiEffectiveRangeData::ValidType_t::DEAD_ALLY:
947  case objects::MiEffectiveRangeData::ValidType_t::DEAD_PARTY:
948  if(!isEnemy)
949  {
950  // Skills that affect parties or dead entities are not
951  // usable by enemies
952  targetType = (int16_t)AI_SKILL_TYPES_ALLY;
953  }
954  break;
955  default:
956  break;
957  }
958 
959  if(targetType != -1)
960  {
961  auto damage = skillData->GetDamage()->GetBattleDamage();
962  switch(damage->GetFormula())
963  {
964  case objects::MiBattleDamageData::Formula_t::DMG_NORMAL:
965  case objects::MiBattleDamageData::Formula_t::DMG_PERCENT:
966  case objects::MiBattleDamageData::Formula_t::DMG_SOURCE_PERCENT:
967  case objects::MiBattleDamageData::Formula_t::DMG_MAX_PERCENT:
968  case objects::MiBattleDamageData::Formula_t::DMG_STATIC:
969  // Do not add skills that damage allies by default
970  if(targetType == (int16_t)AI_SKILL_TYPES_ENEMY)
971  {
972  auto target = skillData->GetTarget();
973  if(target->GetRange() > 0)
974  {
975  skillMap[AI_SKILL_TYPE_CLSR].push_back(skillID);
976  }
977  else
978  {
979  skillMap[AI_SKILL_TYPE_LNGR].push_back(skillID);
980  }
981  }
982  break;
983  case objects::MiBattleDamageData::Formula_t::HEAL_NORMAL:
984  case objects::MiBattleDamageData::Formula_t::HEAL_MAX_PERCENT:
985  case objects::MiBattleDamageData::Formula_t::HEAL_STATIC:
986  {
987  skillMap[AI_SKILL_TYPE_HEAL].push_back(skillID);
988  }
989  break;
990  default:
992  if(targetType == (int16_t)AI_SKILL_TYPE_DEF)
993  {
994  skillMap[AI_SKILL_TYPE_DEF].push_back(skillID);
995  }
996  else
997  {
998  skillMap[AI_SKILL_TYPE_SUPPORT].push_back(skillID);
999  }
1000  break;
1001  }
1002  }
1003  }
1004 
1005  aiState->SetSkillMap(skillMap);
1006  }
