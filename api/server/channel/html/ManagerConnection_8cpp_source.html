67 {
68  const libcomp::Message::ConnectionMessage *cMessage = dynamic_cast<

71  switch(cMessage->GetConnectionMessageType())

73  case libcomp::Message::ConnectionMessageType::CONNECTION_MESSAGE_ENCRYPTED:
74  {
75  const libcomp::Message::Encrypted *encrypted = dynamic_cast<
76  const libcomp::Message::Encrypted*>(cMessage);
77 
78  auto connection = encrypted->GetConnection();
79 
80  if(mWorldConnection == connection)

82  RequestWorldInfo();

85  return true;

91  const libcomp::Message::ConnectionClosed*>(cMessage);

104  server->Shutdown();
105  }
106 
107  return true;
108  }
109  break;
110  default:
111  break;
112  }
113

350 void ManagerConnection::HandleClientTimeouts(uint64_t now, uint16_t timeout)
351 {
352  std::list<libcomp::String> timeOuts;
353  {
354  ServerTime expireBefore = now - (ServerTime)(timeout * 1000000ULL);
355

357  for(auto it = mClientConnections.begin();
358  it != mClientConnections.end(); it++)
359  {

361  if(clientTimeout && clientTimeout <= expireBefore)
362  {
363  timeOuts.push_back(it->first);

365  // Stop the timeout from throwing multiple times
366  it->second->RefreshTimeout(0, 0);
367  }
368  }
369  }

372  {

374  {

376  "Client connection timed out: %1\n").Arg(timedOut));
