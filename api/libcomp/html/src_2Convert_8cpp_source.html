79  // that big.
80  if(0 == size)
81  {

83  }

85  {

87  }

151  // Retrieve the next byte of the string and determine the mapped code
152  // point for the desired encoding. CP932 is a multi-byte format similar
153  // to Shift-JIS. As such, if the most significant bit is set, another
154  // byte needs to be read and added to the code point before conversion.
155  // After each byte read from the string, the string pointer should be
156  // advanced.
157  uint16_t cp932 = *(szString++);
158 
159  // Certain byte values indicate multibyte characters.
160  if( (0x81 <= cp932 && 0x9F >= cp932) ||
161  (0xE0 <= cp932 && 0xFC >= cp932) )
162  {
163  // Sanity check that we can read the 2nd byte of the code point.
164  // If not, we should return an empty string to indicate an error.
167  if(1 > size--)
168  {
169  return String();
170  }
171 
172  // A multi-byte CP932 code point consists of the first byte in the
173  // 8 most significant bits and the second byte in the 8 least
174  // significant bits.
175  cp932 = (uint16_t)( (cp932 << 8) | *(szString++) );
176  }
177 
178  // If there is no mapped codec, return an empty string to indicate an
179  // error.
180  String::CodePoint unicode = pMappingFrom[cp932];
181 
182  if(0 == unicode)
183  {
184  return String();
185  }
