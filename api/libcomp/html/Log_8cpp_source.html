241  "DEBUG: %1",
242  "%1",
243  "WARNING: %1",
244  "ERROR: %1",
245  "CRITICAL: %1",
246  };
247 
248  // Log a critical error message. If the configuration option is true, log
249  // the message to the log file. Regardless, pass the message to all the
250  // log hooks for processing. Critical messages have the text "CRITICAL: "
251  // appended to them.
252  if(0 > level || LOG_LEVEL_COUNT <= level || !mLogEnables[level])
253  return;

257  // Lock the muxtex.
258  std::lock_guard<std::mutex> lock(mLock);
259

261  {
262  auto now = std::chrono::system_clock::now();
263  auto duration = std::chrono::duration_cast< std::chrono::duration<
264  int64_t, std::ratio<86400>> >( now.time_since_epoch() ).count();
265 
266  if(mLogRotationEnabled && mLogRotationDays <= (duration - mLastLog))
267  {
268  RotateLogs();
269  }
270 
271  mLastLog = duration;
272 
273  if(mLogFileTimestampEnabled)
274  {
275  auto currentTime = std::chrono::system_clock::to_time_t(now);
276 
277  std::stringstream ss;
278  ss << std::put_time(std::localtime(&currentTime), "%Y/%m/%d %T");
279 
280  String formattedTime = String("[%1] ").Arg(ss.str());
281 
282  mLogFile->write(formattedTime.C(),
283  (std::streamsize)(formattedTime.Length() * sizeof(char)));
284  }
285

287  (std::streamsize)(final.Length() * sizeof(char)));
288  mLogFile->flush();
289  }

292  for(auto i : mHooks)
293  {
294  (*i.first)(level, final, i.second);

476 void Log::RotateLogs()
477 {
478  // Do not rotate if the main log does not exist.
479  if(!mLogPath.IsEmpty() && !FileExists(mLogPath))
480  {
481  return;
482  }
483 
484  // Delete the last log that we will rotate out.
485  FileDelete(libcomp::String("%1.%2").Arg(mLogPath).Arg(
486  mLogRotationCount));
487  FileDelete(libcomp::String("%1.%2.gz").Arg(mLogPath).Arg(
488  mLogRotationCount));
489 
490  // Now we need to rotate them out moving the files as we go.
491  for(int i = mLogRotationCount - 1; i > 0; --i)
492  {
493  FileMove(libcomp::String("%1.%2").Arg(mLogPath).Arg(i),
494  libcomp::String("%1.%2").Arg(mLogPath).Arg(i + 1));
495  FileMove(libcomp::String("%1.%2.gz").Arg(mLogPath).Arg(i),
496  libcomp::String("%1.%2.gz").Arg(mLogPath).Arg(i + 1));
497  }
498 
499  // Now close the main log and move it.
500  if(mLogFile)
501  {
502  mLogFile->close();
503  delete mLogFile;
504  }
505 
506  FileMove(mLogPath, mLogPath + ".1");
507 
508  // Open the log again.
509  mLogFile = new std::ofstream();
510  mLogFile->open(mLogPath.C(), std::ofstream::out | std::ofstream::trunc);
511  mLogFile->flush();
512 
513  // If this failed, close it.
514  if(!mLogFile->good())
515  {
516  delete mLogFile;
517  mLogFile = nullptr;
518  mLogPath.Clear();
519  }
520 
521  // If compression is enabled compress the log we just moved.
522  if(mLogCompression)
523  {
524  std::thread th([](libcomp::String compressPath)
525  {
526  bool error = false;
527  char buffer[4096];
528  libcomp::String compressedPath = compressPath + ".gz";
529 
530  FILE *in = fopen(compressPath.C(), "rb");
531  gzFile gf = gzopen(compressedPath.C(), "wb");
532 
533  if(in && gf)
534  {
535  while(!feof(in))
536  {
537  auto sz = fread(buffer, 1, sizeof(buffer), in);
538 
539  if(0 >= sz)
540  {
541  break;
542  }
543 
544  if((int)sz != gzwrite(gf, buffer, (unsigned)sz))
545  {
546  error = true;
547  break;
548  }
549  }
550  }
551 
552  gzclose(gf);
553  fclose(in);
554 
555  if(error)
556  {
557  FileDelete(compressedPath);
558  }
559  else
560  {
561  FileDelete(compressPath);
562  }
563  }, mLogPath + ".1");
564 
565  // Detach the thread and let it work on it's own.
566  th.detach();
567  }
568 }
