228 void Log::LogMessage(Log::Level_t level, const String& msg)
229 {
230  // Prepend these to messages.
231  static const String gLogMessages[Log::LOG_LEVEL_COUNT] = {
232  "DEBUG: %1",
233  "%1",
234  "WARNING: %1",
235  "ERROR: %1",
236  "CRITICAL: %1",
237  };
238 
239  // Log a critical error message. If the configuration option is true, log
240  // the message to the log file. Regardless, pass the message to all the
241  // log hooks for processing. Critical messages have the text "CRITICAL: "
242  // appended to them.
243  if(0 > level || LOG_LEVEL_COUNT <= level || !mLogEnables[level])
244  return;
245 
246  String final = String(gLogMessages[level]).Arg(msg);
247 
248  // Lock the muxtex.
249  std::lock_guard<std::mutex> lock(mLock);
250 
251  if(nullptr != mLogFile)
252  {
253  if(mLogFileTimestampEnabled)
254  {
255  auto currentTime = std::chrono::system_clock::to_time_t(
256  std::chrono::system_clock::now());
257 
258  std::stringstream ss;
259  ss << std::put_time(std::localtime(&currentTime), "%Y/%m/%d %T");
260 
261  String formattedTime = String("[%1] ").Arg(ss.str());
262 
263  mLogFile->write(formattedTime.C(),
264  (std::streamsize)(formattedTime.Length() * sizeof(char)));
265  }
266 
267  mLogFile->write(final.C(),
268  (std::streamsize)(final.Length() * sizeof(char)));
269  mLogFile->flush();
270  }
271 
272  // Call all hooks.
273  for(auto i : mHooks)
274  {
275  (*i.first)(level, final, i.second);
276  }
277 
278  // Call all lambda hooks.
279  for(auto func : mLambdaHooks)
280  {
281  func(level, final);
282  }
283 }
