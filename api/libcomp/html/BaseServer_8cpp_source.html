545 bool BaseServer::LoadDataFromFile(const libcomp::String& filePath,
546  PersistentObjectMap& records, bool registerRecords,
547  const std::set<std::string>& specificTypes)
548 {
549  tinyxml2::XMLDocument doc;
550  if(tinyxml2::XML_SUCCESS != doc.LoadFile(filePath.C()))
551  {
552  return false;
553  }
554 
555  LOG_DEBUG(libcomp::String("Loading records from file '%1'...\n").Arg(filePath));
556 
557  const tinyxml2::XMLElement *objXml = doc.RootElement()->FirstChildElement("object");
558 
559  while(nullptr != objXml)
560  {
561  bool typeExists = true;
562 
563  std::string name(objXml->Attribute("name"));
564  if(specificTypes.size() > 0 && specificTypes.find(name) == specificTypes.end())
565  {
566  continue;
567  }
568 
569  auto typeHash = libcomp::PersistentObject::GetTypeHashByName(name, typeExists);
570  if(!typeExists)

572  return false;
573  }
574 
575  // Retrieve the UID if specified
576  libobjgen::UUID uuid;
577  const tinyxml2::XMLElement *memberXml = objXml->FirstChildElement("member");

581  {
582  std::string uuidText(memberXml->GetText());
583  uuid = libobjgen::UUID(uuidText);
584  if(uuid.IsNull())
585  {
586  return false;
587  }
588  break;
589  }
590  memberXml = memberXml->NextSiblingElement("member");
591  }
592 
593  auto record = libcomp::PersistentObject::New(typeHash);
594  if(!record->Load(doc, *objXml))
595  {
596  return false;
597  }
598 
599  if(name == "Account")
600  {
601  // Check that there is a password and salt it
602  auto account = std::dynamic_pointer_cast<objects::Account>(record);
603 
604  if(account->GetUsername().IsEmpty() ||

611

613  account->SetSalt(salt);
614  account->SetPassword(libcomp::Decrypt::HashPassword(
615  account->GetPassword(), salt));
616  }
617

619  {
620  return false;
621  }

623  {

625  }
