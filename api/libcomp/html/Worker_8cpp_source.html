104  libcomp::Message::Shutdown*>(pMessage);
105 
106  // Check for an execute message.
107  libcomp::Message::Execute *pExecute = dynamic_cast<
108  libcomp::Message::Execute*>(pMessage);
109 
110  // Do not handle any more messages if a shutdown was sent.
111  if(nullptr != pShutdown || !mRunning)
112  {
113  mRunning = false;
114  }
115  else if(nullptr != pExecute)
116  {
117  // Run the code now.

120  else

123  auto it = mManagers.find(pMessage->GetType());
124

126  if(it == mManagers.end())

130  }
