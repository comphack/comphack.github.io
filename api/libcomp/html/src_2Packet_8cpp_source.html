120 void Packet::GrowPacket(uint32_t sz)
121 {
122  // Allocate the packet data (if needed).
123  Allocate();
124 
125  // Make sure the packet is growing.
126  if(0 == sz)
127  {
128  PACKET_EXCEPTION("Attempted to grow the packet by 0 bytes", this);
129  }
130 
131  // Calculate how big the packet will be if it can grow.
132  uint32_t newSize = mPosition + sz;
133 
134  /* If the new size of the packet is less than the current size, keep the
135  * packet at it's current size. If the new packet size is too big, throw a
136  * packet exception. If the size is within the max packet size, set the new
137  * size of the packet.
138  */
139  if(newSize < mSize)
140  {
141  // Keep the current size, do nothing.
142  }
143  else if(MAX_PACKET_SIZE < newSize || MAX_PACKET_SIZE < (mPosition + sz))
144  {
145  // The new size is too big, throw an exception.
146  PACKET_EXCEPTION(String("Attempted to grow packet by %1 bytes; "
147  "however, doing so would exceed MAX_PACKET_SIZE").Arg(sz), this);
148  }
149  else
150  {
151  // The new packet size is valid, set it.
152  mSize = newSize;
153  }
154 }
