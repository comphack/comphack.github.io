100 bool PersistentObject::Register(const std::shared_ptr<PersistentObject>& self,
101  const libobjgen::UUID& pUuid)
102 {
103  if(!self->IsDeleted())
104  {
105  bool registered = false;
106 
107  libobjgen::UUID& uuid = self->mUUID;
108 
109  std::lock_guard<std::mutex> lock(mCacheLock);
110  if(!pUuid.IsNull() && !uuid.IsNull())
111  {
112  // Unregister old UUID, keep if making a copy
113  auto it = sCached.find(uuid.ToString());
114  if(it != sCached.end() && it->second.lock() == self)
115  {
116  sCached.erase(it);
117  }
118  }
119 
120  if(!pUuid.IsNull())
121  {
122  uuid = pUuid;
123  }
124  else if(uuid.IsNull())
125  {
126  uuid = libobjgen::UUID::Random();
127  registered = true;
128  }
129 
130  if(!registered && sCached.find(uuid.ToString()) == sCached.end())
131  {
132  registered = true;
133  }
134 
135  std::string uuidString = uuid.ToString();
136 
137  if(registered)
138  {
139  self->mSelf = self;
140  sCached[uuidString] = self;
141 
142  return true;
143  }
144  else
145  {
146  LOG_ERROR(String("Duplicate object detected: %1\n").Arg(
147  uuidString));
148  }
149  }
150 
151  return false;
152 }
