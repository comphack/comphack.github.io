854  bool LoadBinaryData(gsl::not_null<DataStore*> pDataStore,
855  const libcomp::String& binaryFile, bool decrypt,
856  uint16_t tablesExpected, std::list<std::shared_ptr<T>>& records,
857  bool printResults = true)
858  {

860 
861  auto path = libcomp::String("/BinaryData/") + binaryFile;
862 
863  if(decrypt)
864  {
865  data = pDataStore->DecryptFile(path);
866  }
867  else
868  {
869  data = pDataStore->ReadFile(path);
870  }
871 
872  if(data.empty())
873  {
874  if(printResults)
875  {
876  PrintLoadResult(binaryFile, false, 0, 0);
877  }
878  return false;
879  }
880 
881  std::stringstream ss(std::string(data.begin(), data.end()));
882  libcomp::ObjectInStream ois(ss);
883 
884  uint16_t entryCount, tableCount;
885  if(!LoadBinaryDataHeader(ois, binaryFile, tablesExpected,
886  entryCount, tableCount))
887  {
888  return false;
889  }
890 
891  size_t dynamicCounts = (size_t)(entryCount * tableCount);
892  for(uint16_t i = 0; i < dynamicCounts; i++)
893  {
894  uint16_t ds;
895  ois.stream.read(reinterpret_cast<char*>(&ds), sizeof(ds));
896  ois.dynamicSizes.push_back(ds);
897  }
898 
899  for(uint16_t i = 0; i < entryCount; i++)
900  {
901  auto entry = std::shared_ptr<T>(new T);
902 
903  if(!entry->Load(ois))
904  {
905  if(printResults)
906  {
907  PrintLoadResult(binaryFile, false, entryCount,
908  records.size());
909  }
910  return false;
911  }
912 
913  records.push_back(entry);
914  }
915 
916  bool success = entryCount == records.size() && ois.stream.good();
917  if(printResults)
918  {
919  PrintLoadResult(binaryFile, success, entryCount, records.size());
920  }
921 
922  return success;
923  }
