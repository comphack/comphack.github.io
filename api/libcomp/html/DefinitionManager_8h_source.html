760  }

762  if(data.empty())

772  libcomp::ObjectInStream ois(ss);
773 
774  uint16_t entryCount, tableCount;

783  {
784  uint16_t ds;
785  ois.stream.read(reinterpret_cast<char*>(&ds), sizeof(ds));
786  ois.dynamicSizes.push_back(ds);
787  }
788 
789  for(uint16_t i = 0; i < entryCount; i++)
790  {
791  auto entry = std::shared_ptr<T>(new T);
792 
793  if(!entry->Load(ois))
794  {
795  if(printResults)
796  {
797  PrintLoadResult(binaryFile, false, entryCount, records.size());
798  }
799  return false;
800  }
801 
802  records.push_back(entry);
803  }
804 
805  bool success = entryCount == records.size() && ois.stream.good();
806  if(printResults)

809  }
